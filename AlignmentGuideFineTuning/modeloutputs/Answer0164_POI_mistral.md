PROGRAM FahrenheitToCelsius

VAR_INPUT
    Fahrenheit : REAL;
END_VAR

VAR_OUTPUT
    Celsius : REAL;
    ValidInput : BOOL;
END_VAR

BEGIN
    IF Fahrenheit < -459.67 THEN
        Celsius := 0;
        ValidInput := FALSE;
    ELSE
        Celsius := (Fahrenheit - 32) * 5 / 9;
