    Start : BOOL;

    PresetTime : TIME;

    Running : BOOL;

    StartTime : TIME;
    ElapsedTime : TIME;

METHOD Execute : BOOL
BEGIN
    // Reset outputs
    Running := FALSE;
    Done := FALSE;
    RemainingTime := TIME(0);
    ElapsedTime := TIME(0);

    // Check if timer is running
    IF Start THEN
        // Store current time
        StartTime := CURRENT_TIME;
        Running := TRUE;
    END_IF;

    // Check if timer is stopped
    IF Stop THEN
        Running := FALSE;
        Done := TRUE;
    END_IF;

    // Calculate elapsed time
    IF Running THEN
        ElapsedTime := CURRENT_TIME - StartTime;
        IF ElapsedTime >= PresetTime THEN
            Done := TRUE;
        END_IF;

        // Calculate remaining time
        RemainingTime := PresetTime - ElapsedTime;
    END_IF;

    RETURN TRUE;
END_METHOD

    Done := FALSE;
    RemainingTime := TIME(0);
